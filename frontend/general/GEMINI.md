```javascript
// Number formatting with thousands separator
function formatNumber(number, decimals = 2) {
    return new Intl.NumberFormat('en-US', {
        minimumFractionDigits: 0,
        maximumFractionDigits: decimals
    }).format(number);
}

// Fraction simplification
function simplifyFraction(numerator, denominator) {
    const gcd = (a, b) => b === 0 ? a : gcd(b, a % b);
    const divisor = gcd(numerator, denominator);
    return {
        numerator: numerator / divisor,
        denominator: denominator / divisor
    };
}

// Ratio simplification
function simplifyRatio(...parts) {
    const gcd = (a, b) => b === 0 ? a : gcd(b, a % b);
    const divisor = parts.reduce((acc, val) => gcd(acc, val));
    return parts.map(part => part / divisor);
}
```

## Development Best Practices

### Code Organization
```javascript
// Separate calculation logic from UI
class PercentageCalculator extends CalculatorBase {
    // Pure calculation methods (no DOM manipulation)
    calculatePercentage(value, percentage) {
        return (value * percentage) / 100;
    }

    // UI-specific methods
    displayResult(result) {
        // DOM manipulation here
    }
}
```

### Error Handling
```javascript
try {
    const result = this.calculate();
    this.displayResult(result);
} catch (error) {
    if (error instanceof DivisionByZeroError) {
        this.showError('Cannot divide by zero');
    } else if (error instanceof InvalidInputError) {
        this.showError('Please enter valid numbers');
    } else {
        this.showError('An error occurred. Please try again.');
    }
}
```
